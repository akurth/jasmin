77  
78      private String js(String ... sources) throws IOException {
79          return minimize(MimeType.JS, sources);
80      }
81  
82      private String css(String ... sources) throws IOException {
83          return minimize(MimeType.CSS, sources);
84      }
85  
86      private String minimize(MimeType type, String ... sources) throws IOException {
87          Node dest;
88          References references;
89  
90          dest = world.memoryNode();
91          try (Writer writer = dest.createWriter()) {
92              references = new References(type, true);
93              for (String source : sources) {
94                  references.add(true, world.memoryNode(source));
95              }
96              references.writeTo(writer);
97          }
98          return dest.readString();
99      }
