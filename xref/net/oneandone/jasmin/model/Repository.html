128     public List<String> sequence(String... names) throws CyclicDependency {
129         List<Module> moduleList;
130         List<String> result;
131 
132         moduleList = new ArrayList<Module>();
133         for (String name : names) {
134             moduleList.add(get(name));
135         }
136         result = new ArrayList<String>();
137         for (Module module : sequence(moduleList)) {
138             result.add(module.getName());
139         }
140         return result;
141     }

144     // topological sort
145     public List<Module> sequence(List<Module> moduleList) throws CyclicDependency {
146         List<Module> work;
147         Graph<Module> graph;
148         Module left;
149         Module right;
150 
151         graph = new Graph<Module>();
152         for (int i = 0; i < moduleList.size(); i++) {
153             right = moduleList.get(i);
154             graph.addNode(right);
155             if (i > 0) { // inject dependency to previous modules
156                 left = moduleList.get(i - 1);
157                 graph.addEdge(left, right);
158             }
159         }
160         work = new ArrayList<Module>();
161         work.addAll(moduleList);
162         for (int i = 0; i < work.size(); i++) { // list grows!
163             right = work.get(i);
164             for (Module l : right.dependencies()) {
165                 graph.addEdge(l, right);
166                 if (!work.contains(l)) {
167                     work.add(l);
168                 }
169             }
170         }
171         return graph.sort();
172     }

176     public References resolve(Request request) throws IOException, CyclicDependency {
177         List<Module> includes;
178         List<Module> excludes;
179         References references;
180         List<Module> moduleList;
181         Node resolved;
182         boolean minimize;
183 
184         includes = new ArrayList<>();
185         excludes = new ArrayList<>();
186         for (String name : Module.SEP.split(request.modules)) {
187             if (name.length() == 0) {
188                 throw new IllegalStateException();
189             }
190             if (name.charAt(0) == Module.NOT) {
191                 excludes.add(get(name.substring(1)));
192             } else {
193                 includes.add(get(name));
194             }
195         }
196         moduleList = sequence(includes);
197         moduleList.removeAll(sequence(excludes));
198         references = new References(request.type, request.minimize);
199         for (Module module : moduleList) {
200             for (File file : module.resolve(request)) {
201                 resolved = file.get(request.minimize);
202                 minimize = request.minimize && file.getMinimized() == null;
203                 references.add(minimize, resolved);
204             }
205         }
206         return references;
207     }

209     public List<String> getVariants() {
210         List<String> results;
211         String variant;
212 
213         results = new ArrayList<>();
214         for (Module module : modules) {
215             for (File file : module.files()) {

217                 if (variant != null && !results.contains(variant)) {

219                 }
220             }
221         }

227     /** Loads modules from classpath */
228     public void loadClasspath(Resolver resolver) throws IOException {
229         Enumeration<URL> e;
230 
231         e = getClass().getClassLoader().getResources(MODULE_DESCRIPTOR);
232         while (e.hasMoreElements()) {
233             loadModule(resolver, e.nextElement());
234         }
235     }

237     private void loadModule(Resolver resolver, URL url) throws IOException {
238         FileNode classpathItem;
239         Node descriptor;
240         Node properties;
241         Node base;
242 
243         classpathItem = resolver.getWorld().locateClasspathItem(url, "/" + MODULE_DESCRIPTOR);
244         base = classpathItem.isFile() ? classpathItem.openZip() : classpathItem;
245         descriptor = resolver.resolve(base, MODULE_DESCRIPTOR);
246         properties = resolver.resolve(base, MODULE_PROPERTIES);
247         loadLibrary(resolver, base, descriptor, properties);
248     }

262     /** Core method for loading. A library is a module or an application */
263     public void loadLibrary(Resolver resolver, Node base, Node descriptor, Node properties) throws IOException {
264         Source source;
265         Module module;
266         Library library;
267         File file;
268 
269         addReload(descriptor);
270         source = Source.load(properties, base);
271         library = (Library) Library.TYPE.loadXml(descriptor).get();
272         autoFiles(resolver, library, source);
273         for (net.oneandone.jasmin.descriptor.Module descriptorModule : library.modules()) {
274             module = new Module(descriptorModule.getName(), source);
275             notLinked.put(module, descriptorModule.dependencies());
276             for (Resource resource : descriptorModule.resources()) {
277                 file = resolver.resolve(source.classpathBase, resource);
278                 addReload(file);
279                 resolver.resolve(source.classpathBase, resource);
280                 module.files().add(file);
281             }
282             add(module);
283         }
284     }

291     private void addReload(Node node) {
292         if (node instanceof WebdavNode) {
293             return;
294         }
295         if (node instanceof FileNode) {
296             // done
297         } else if (node instanceof ZipNode) {
298             node = node.getWorld().file(((ZipNode) node).getRoot().getZip().getName());
299         } else if (node instanceof MemoryNode) {
300             // for tests
301         } else {
302             throw new IllegalStateException("unexpected node: " + node);
303         }
304         if (!reloadFiles.contains(node)) {
305             reloadFiles.add(node);
306         }
307     }

311         autoFiles(life, source, MimeType.CSS, library.csss());

314     private void autoFiles(Resolver resolver, Source source, MimeType type, List<String> includes) throws IOException {
315         Filter filter;
316         Module module;
317         Ref ref;
318         File file;
319         List<String> depends;
320         List<String> calls;
321         List<Ref> refs;
322         Map<String, List<Ref>> map;
323         Ref normal;
324         Ref companion;
325         Ref minimized;
326 
327         filter =  new Filter();
328         filter.include(includes);
329         map = new HashMap<String, List<Ref>>();
330         for (Node node : source.classpathBase.find(filter)) {
331             ref = Ref.create(node, source);
332             refs = map.get(ref.module);
333             if (refs == null) {
334                 refs = new ArrayList<Ref>();
335                 map.put(ref.module, refs);
336             }
337             refs.add(ref);
338         }
339         for (Map.Entry<String, List<Ref>> entry : map.entrySet()) {
340             module = lookup(entry.getKey());
341             if (module == null) {
342                 module = new Module(entry.getKey(), source);
343                 modules.add(module);
344             } else {
345                 if (module.getSource() != source) {
346                     throw new IllegalStateException();
347                 }
348             }
349             refs = entry.getValue();
350             while (!refs.isEmpty()) {
351                 normal = refs.remove(0);
352                 companion = removeCompanionOpt(refs, normal);
353                 if (companion == null) {
354                     if (normal.minimized) {
355                         minimized = normal;
356                     } else {
357                         minimized = null;
358                     }
359                 } else {
360                     if (normal.minimized) {
361                         minimized = normal;
362                         normal = companion;
363                     } else {
364                         minimized = companion;
365                     }
366                 }
367                 file = resolver.resolve(source.classpathBase, new Resource(type, Base.CLASSPATH, normal.path,
368                         minimized != null ? minimized.path : null, normal.variant));
369                 try {
370                     depends = notLinked.get(module);
371                     if (depends == null) {
372                         depends = new ArrayList<String>();
373                         notLinked.put(module, depends);
374                     }
375                     calls = new ArrayList<String>();
376                     Parser.parseComment(file.getNormal().readString(), depends, calls);
377                 } catch (IOException e) {
378                     throw new IOException(normal.node.getURI() + ": " + e.getMessage(), e);
379                 }
380                 if (calls.size() > 0) {
381                     if (file.getMinimized() != null) {
382                         throw new UnsupportedOperationException(file.getNormal().getURI().toString());
383                     }
384                     Node tmp = resolver.getWorld().memoryNode();
385                     try (OutputStream dest = tmp.createOutputStream()) {
386                         for (String webservice : calls) {
387                             Call.call(attributes, webservice, dest);
388                             dest.write('\n');
389                         }
390                         try (InputStream orig = file.getNormal().createInputStream()) {
391                             resolver.getWorld().getBuffer().copy(orig, dest);
392                         }
393                     }
394                     file = new File(tmp, null, file.getType(), file.getVariant());
395                 }
396                 module.files().add(file);
397                 addReload(file);
398             }
399         }
400     }

404     // companion: Ref with same module and variant, but different minimization
405     private static Ref removeCompanionOpt(List<Ref> refs, Ref cmp) {
406         Iterator<Ref> iter;
407         Ref ref;
408 
409         iter = refs.iterator();
410         while (iter.hasNext()) {
411             ref = iter.next();
412             if (ref.module.equals(cmp.module) && Util.eq(ref.variant, cmp.variant)) {
413                 iter.remove();
414                 return ref;
415             }
416         }
417         return null;
418     }

420     private static class Ref {
421         private static final String MIN = "-min";
422         public static Ref create(Node node, Source source) {
423             String path;
424             String module;
425             int idx;
426             boolean minimized;
427             String variant;
428 
429             path = node.getRelative(source.classpathBase);
430             module = path;
431             module = Strings.removeLeftOpt(module, "script/");
432             module = Strings.removeLeftOpt(module, "style/");
433             idx = module.lastIndexOf('.');
434             if (idx != -1) {
435                 module = module.substring(0, idx);
436             }
437             minimized = module.endsWith(MIN);
438             if (minimized) {
439                 module = module.substring(0, module.length() - MIN.length());
440             }
441             idx = module.indexOf(':');
442             if (idx != -1) {
443                 variant = module.substring(idx + 1);
444                 module = module.substring(0, idx);
445             } else {
446                 variant = null;
447             }
448             module = source.artifactId + "-" + module.replace('/', '-');
449             return new Ref(node, path, module, variant, minimized);
450         }

452         public final Node node;
453         public final String path;
454         public final String module;
455         public final String variant;
456         public final boolean minimized;
457 
458         public Ref(Node node, String path, String module, String variant, boolean minimized) {
459             this.node = node;
460             this.path = path;
461             this.module = module;
462             this.variant = variant;
463             this.minimized = minimized;
464         }

469     /**
470      * Call this after you've loaded all libraries
471      * @return reload files
472      */
473     public List<Node> link() {
474         List<Module> dependencies;
475         Module module;
476         Module resolved;
477         StringBuilder problems;
478         List<Node> result;
479 
480         problems = new StringBuilder();
481         for (Map.Entry<Module, List<String>> entry : notLinked.entrySet()) {
482             module = entry.getKey();
483             dependencies = module.dependencies();
484             for (String name : entry.getValue()) {
485                 resolved = lookup(name);
486                 if (resolved == null) {
487                     problems.append("module '" + module.getName() + "': cannot resolve dependency '" + name + "'\n");
488                 } else {
489                     dependencies.add(resolved);
490                 }
491             }
492         }
493         if (problems.length() > 0) {
494             throw new IllegalArgumentException(problems.toString());
495         }
496         result = reloadFiles;
497         notLinked = null;
498         reloadFiles = null;
499         return result;
500     }
