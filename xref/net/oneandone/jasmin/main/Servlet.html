123     /** Creates engine from configuration and resolve. Sychronized ensures we initialize only once. */
124     private synchronized void lazyInit(HttpServletRequest request) throws IOException {
125         List<File> files;
126         URL url;
127         long lastModified;
128         long now;
129         Resolver resolver;
130         Node localhost;
131         FileNode file;
132         Object[] tmp;
133 
134         resolver = application.resolver;
135         if (engine != null && resolver.isLife()) {
136             for (Node node : reloadFiles) {
137                 lastModified = node.getLastModified();
138                 if (lastModified > loaded) {
139                     now = System.currentTimeMillis();
140                     if (lastModified > now) {
141                         // fail to avoid repeated re-init
142                         throw new IOException(node.getURI() + " has lastModifiedDate in the future: "
143                                 + new Date(lastModified) + "(now: " + new Date(now) + ")");
144                     }
145                     LOG.info("reloading jasmin for application '" + application.getContextPath() + "' - changed file: " + node);
146                     engine = null;
147                     resolver.reset();
148                 }
149             }
150         }
151         if (engine == null) {
152             url = new URL(request.getRequestURL().toString());
153             try {
154                 // always use http, avoid https
155                 localhost = resolver.getWorld().node(new URI("http", null, url.getHost(), url.getPort(), "", null, null));
156             } catch (URISyntaxException e) {
157                 throw new IllegalStateException(e);
158             }
159             tmp = application.createEngine(docroot, localhost);
160             engine = (Engine) tmp[0];
161             LOG.info("started engine, initial url=" + url);
162             for (Module module : engine.repository.modules()) {
163                 files = module.files();
164                 if (files.size() > 1) {
165                     LOG.warn("deprecated: module '" + module.getName() + "' contains more than 1 file: " + files);
166                 }
167                 for (File f : files) {
168                     if (f.getNormal() instanceof WebdavNode) {
169                         LOG.warn("deprecated: module '" + module.getName() + "' uses base LOCALHOST: " + f.getNormal().getURI());
170                     }
171                 }
172             }
173             if (resolver.isLife()) {
174                 reloadFiles = (List<Node>) tmp[1];
175                 file = resolver.getLiveXml();
176                 if (file != null) {
177                     reloadFiles.add(file);
178                 }
179                 LOG.info("reload if one of these " + reloadFiles.size() + " files is modified: ");
180                 for (Node node : reloadFiles) {
181                     LOG.info("  " + node.getURI());
182                 }
183                 loaded = System.currentTimeMillis();
184             }
185             if (engine == null) {
186                 throw new IllegalStateException();
187             }
188         }
189     }

193     /**
194      * Called by the servlet engine to process get requests:
195      * a) to set the Last-Modified header in the response
196      * b) to check if 304 can be redurned if the "if-modified-since" request header is present
197      * @return -1 for when unknown
198      */
199     @Override
200     public long getLastModified(HttpServletRequest request) {
201         String path;
202         int idx;
203         long result;
204 
205         result = -1;
206         try {
207             path = request.getPathInfo();
208             if (path != null && path.startsWith("/get/")) {
209                 lazyInit(request);
210                 path = path.substring(5);
211                 idx = path.indexOf('/');
212                 if (idx != -1) {
213                     path = path.substring(idx + 1);
214                     result = engine.getLastModified(path);
215                 }
216             }
217         } catch (IOException e) {
218             error(request, "getLastModified", e);
219             // fall-through
220         } catch (RuntimeException | Error e) {
221             error(request, "getLastModified", e);
222             throw e;
223         }
224         LOG.debug("getLastModified(" + request.getPathInfo() + ") -> " + result);
225         return result;
226     }

230         try {
231             doGetUnchecked(request, response);

233             // I can't compile against this class because the servlet api does not officially
234             // report this situation ...

236             if (e.getClass().getName().equals("org.apache.catalina.connector.ClientAbortException")) {
237                 // this is not an error: the client browser closed the response stream, e.g. because
238                 // the user already left the page
239                 LOG.info("aborted by client", e);
240             } else {
241                 error(request, "get", e);
242             }
243             throw e;
244         } catch (RuntimeException | Error e) {
245             error(request, "get", e);
246             throw e;
247         }

252     private void doGetUnchecked(HttpServletRequest request, HttpServletResponse response) throws IOException {
253         String path;
254 
255         path = request.getPathInfo();
256         if (path == null) {
257             response.sendRedirect(request.getContextPath() + request.getServletPath() + "/");
258             return;
259         }
260         lazyInit(request);
261         LOG.debug("get " + path);
262         if (path.startsWith("/get/")) {
263             get(request, response, path.substring(5));
264             return;
265         }
266         if (path.equals("/admin/")) {
267             main(response);
268             return;
269         }
270         if (path.equals("/admin/repository")) {
271             repository(request, response);
272             return;
273         }
274         if (path.equals("/admin/hashCache")) {
275             hashCache(response);
276             return;
277         }
278         if (path.equals("/admin/contentCache")) {
279             contentCache(response);
280             return;
281         }
282         if (path.startsWith(MODULE_PREFIX)) {
283             module(request, response, path.substring(MODULE_PREFIX.length()));
284             return;
285         }
286         if (path.equals("/admin/reload")) {
287             reload(response);
288             return;
289         }
290         if (path.equals("/admin/check")) {
291             fileCheck(response);
292             return;
293         }
294         notFound(request, response);
295     }

297     private static final long FIVE_MINUTES = 1000L * 60 * 5;
298     private static final long SEVEN_DAYS = 1000L * 3600 * 24 * 7;
299     private static final long TEN_YEARS = 1000L * 3600 * 24 * 365 * 10;

301     private void get(HttpServletRequest request, HttpServletResponse response, String path) throws IOException {
302         String version;
303         boolean expire;
304         int idx;
305         long started;
306         long duration;
307         int bytes;
308         boolean gzip;
309         long date;
310 
311         if ("/xml/jasmin".equals(request.getServletPath())) {
312             LOG.warn("deprecated servlet path '/xml/jasmin' - use '/jasmin' instead: " + path);
313         }
314         idx = path.indexOf('/');
315         if (idx == -1) {
316             notFound(request, response);
317             return;
318         }
319         started = System.currentTimeMillis();
320         version = path.substring(0, idx);
321         expire = !"no-expires".equals(version);
322         if (expire && !VM_STARTUP_STR.equals(version)) {
323             try {
324                 synchronized (FMT) {
325                     date = FMT.parse(version).getTime();
326                 }
327             } catch (ParseException e) {
328                 notFound(request, response);
329                 return;
330             }
331             if (sameTime(VM_STARTUP, date) || sameTime(otherVmStartupDate, date)) {
332                 // ok
333             } else if (date > otherVmStartupDate) {
334                 otherVmStartupDate = date;
335             } else {
336                 // usually, otherVmStartupDate is smaller, but after switching back, VM_STARTUP will be smaller
337                 if (Math.min(otherVmStartupDate, VM_STARTUP) - date > SEVEN_DAYS) {
338                     gone(request, response);
339                     return;
340                 }
341             }
342         }
343         path = path.substring(idx + 1);
344         if (application.resolver.isLife()) {
345             // unknown headers are ok: see http://tools.ietf.org/html/rfc2616#section-7.1
346             response.addHeader("Hi", "Sie werden bedient von Jasmin, vielen Dank fuer ihren Request!");
347         }
348         checkCharset(request.getHeader("Accept-Charset"));
349         if (expire && application.expires != null) {
350             response.setDateHeader("Expires", started + 1000L * application.expires);
351             response.addHeader("Cache-Control", "max-age=" + application.expires);
352         }
353         gzip = canGzip(request);
354         bytes = engine.process(path, response, gzip);
355         duration = System.currentTimeMillis() - started;
356         LOG.info(path + "|" + bytes + "|" + duration + "|" + gzip + "|" + referer(request));
357     }

359     private static boolean sameTime(long left, long right) {
360         long diff;
361 
362         diff = left - right;
363         if (diff < 0) {
364             diff = -diff;
365         }
366         return diff < FIVE_MINUTES;
367     }

369     private static boolean canGzip(HttpServletRequest request) {
370         String accepted;
371         String userAgent;
372 
373         accepted = request.getHeader("Accept-Encoding");
374         if (accepted == null) {
375             return false;
376         }
377         if (!contains(accepted, "gzip")) {
378             return false;
379         }
380         userAgent = request.getHeader("User-Agent");
381         if (userAgent == null) {
382             LOG.info("unknown user-agent");
383             return false;
384         }
385         if (blacklisted(userAgent)) {
386             LOG.info("user-agent blacklisted: " + userAgent);
387             return false;
388         }
389         return true;
390     }

402     private static Integer versionOf(String str, Pattern client) {
403         String version;
404         Matcher matcher;
405 
406         matcher = client.matcher(str);
407         if (matcher.matches()) {
408             version = matcher.group(1);
409             return new Integer(version);

411             return null;

415     // see http://www.w3.org/Protocols/rfc2616/rfc2616-sec14.html
416     public static void checkCharset(String accepts) throws IOException {
417         if (accepts == null) {
418             // ie7 does not send this header
419             return;
420         }
421         // I've seen both "utf-8" and "UTF-8" -> test case-insensitive
422         if (contains(accepts.toLowerCase(), "utf-8")) {
423             return;
424         }
425         if (contains(accepts, "*")) {
426             return;
427         }
428         throw new IOException("utf-8 is not accepted: " + accepts);
429     }

431     // see http://www.w3.org/Protocols/rfc2616/rfc2616-sec14.html
432     public static boolean contains(String list, String keyword) {
433         int idx;
434         int colon;
435         String quality;
436 
437         idx = list.indexOf(keyword);
438         if (idx == -1) {
439             return false;
440         }
441         idx += keyword.length();
442         colon = list.indexOf(",", idx);
443         if (colon == -1) {
444             colon = list.length();
445         }
446         quality = list.substring(idx, colon);
447         idx = quality.indexOf('=');
448         if (idx == -1) {
449             return true;
450         }
451         return !"0".equals(quality.substring(idx + 1).trim());
452     }

475     private void repository(HttpServletRequest request, HttpServletResponse response) throws IOException {
476         JSONWriter dest;
477         List<Module> modules;
478 
479         response.setContentType("application/json");
480         try (Writer writer = response.getWriter()) {
481             dest = new JSONWriter(writer);
482             dest.array();
483             modules = new ArrayList<Module>(engine.repository.modules());
484             Collections.sort(modules, new Comparator<Module>() {
485                 @Override
486                 public int compare(Module left, Module right) {
487                     return left.getName().compareTo(right.getName());
488                 }
489             });
490             for (Module module : modules) {
491                 dest.object();
492                 dest.key("name");
493                 dest.value(module.getName());
494                 dest.key("details");
495                 dest.value(Strings.removeRight(request.getRequestURL().toString(), "/repository") + "/module/" + module.getName());
496                 dest.endObject();
497             }
498             dest.endArray();
499         } catch (JSONException e) {
500             throw new IllegalStateException(e);
501         }
502     }

513     private void module(HttpServletRequest request, HttpServletResponse response, String name) throws IOException {
514         JSONWriter dest;
515         Module module;
516         Source source;
517 
518         module = engine.repository.lookup(name);
519         source = module.getSource();
520         if (module == null) {
521             notFound(request, response);
522             return;
523         }
524         response.setContentType("application/json");
525         try (Writer writer = response.getWriter()) {
526             dest = new JSONWriter(writer);
527             dest.object();
528             dest.key("name");
529             dest.value(module.getName());
530             dest.key("files");
531             dest.array();
532             for (File file : module.files()) {
533                 dest.object();
534                 dest.key("type");
535                 dest.value(file.getType());
536                 dest.key("normal");
537                 dest.value(file.getNormal().getURI());
538                 if (file.getMinimized() != null) {
539                     dest.key("minimized");
540                     dest.value(file.getMinimized().getURI());
541                 }
542                 dest.key("variant");
543                 dest.value(file.getVariant());
544                 dest.endObject();
545             }
546             dest.endArray();
547             dest.key("dependencies");
548             dest.array();
549             for (Module dependency : module.dependencies()) {
550                 dest.value(dependency.getName());
551             }
552             dest.endArray();
553             dest.key("source");
554             dest.object();
555             dest.key("artifactId");
556             dest.value(source.artifactId);
557             dest.key("groupId");
558             dest.value(source.groupId);
559             dest.key("version");
560             dest.value(source.version);
561             dest.key("scm");
562             dest.value(source.scm);
563             dest.endObject();
564             dest.endObject();

566             throw new IllegalStateException(e);
